	title = GOT_WHITE_WALKER.1.title
	desc = GOT_WHITE_WALKER.1.desc
	picture = GOOD_WITH_MONARCH_eventPicture
	
	major = yes
    
    trigger = {
        NOT = { has_global_flag = white_walkers_empowered }
        tag = WIT
        OR = {
            ai = no # Player can do whatever
            
            # If any nation gets near them before 280
            any_country = {
                is_neighbor_of = ROOT
            }
            
            is_year = 280
        }
    }
    
    immediate = {
        hidden_effect = { set_global_flag = white_walkers_empowered }
    }
    
    mean_time_to_happen = {
		months = 1
    }
    
	option = {
		name = GOT_WHITE_WALKER.1.option.a
        
        remove_country_modifier = summer_reigns
        
        capital_scope = {
            set_base_tax = 10
            set_base_production = 10

            cede_province = WIT
        }
        
        westeros = {
            add_claim = WIT

        }
        westeros_sea_region = {
            discover_country = WIT
        }
	}
}
