# Bountiful Harvest
# Our fields have been blessed, and our harvest is bountiful this year. Grain will be especially valuable this year.
# Wonderful
country_event = {
	id = GOT_TRADEGOODS.1
	title = "GOT_TRADEGOODS.1.title"
	desc = "GOT_TRADEGOODS.1.desc"
	picture = TRADEGOODS_eventPicture
	
	major = yes
	is_triggered_only = yes

	trigger = {
		grain = 1
		NOT = {
			any_country = {
				has_country_modifier = grain_timer
			}
		}
	}
	
	immediate = {
		hidden_effect = {
			add_country_modifier = {
				name = grain_timer
				duration = 1825
				hidden = yes
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = "GOT_TRADEGOODS.1.option.a"
		
		change_price = {
			trade_goods = grain
			key = got_grain_boost
			value = 0.25
			duration = 1825
		}
	}
}
