namespace = got_institutions

# Rumours of Magic
country_event = {
	id = got_institutions.100
	title = got_institutions.100.title
	desc = got_institutions.100.desc
	picture = PAINTER_ARTIST_eventPicture

	is_triggered_only = yes
	
	trigger = {
		always = yes		
	}

	mean_time_to_happen = {
		days = 1
	}

	option = {
		name = got_institutions.100.option.a
		
		add_adm_power = 100
		add_dip_power = 100
		add_mil_power = 100
		add_prestige = 20

		custom_tooltip = got_institution_rumours_penalty
		FROM = {
			add_permanent_province_modifier = {
				name = "rumours_of_magic"
				duration = -1
			}
		}
		hidden_effect = {
			every_country = {
				limit = {
					ai = no
					NOT = { tag = ROOT }
					knows_country = ROOT
				}
				country_event = { id = got_institutions.101 }
			}
			every_country = {
				limit = {
					ai = no
					NOT = { tag = ROOT }
					NOT = { knows_country = ROOT }
				}
				country_event = { id = got_institutions.102 }
			}
		}
	}
}
