namespace = setup

country_event = {
	id = setup.1
	title = setup.1.title
	desc = setup.1.desc
	picture = SPY_eventPicture
	
	is_triggered_only = yes
	hidden = yes
    
    trigger = {
        NOT = { has_global_flag = completed_setup }
    }
	
    immediate = {
        set_global_flag = completed_setup
    }
    
	option = {
		name = setup.1.option.a
		
        #set_global_flag = enable_onaction_logging
        #set_global_flag = enable_event_logging
        
        if = { limit = { has_global_flag = enable_event_logging } log = "[Root.GetName] - setup.1.option.a" }
        
        hre_inheritable = yes
        
        # Add Westeros to HRE
        region_the_north = {
            set_in_empire = yes
        }
        region_riverlands = {
            set_in_empire = yes
        }
        region_crownlands = {
            set_in_empire = yes
        }
        region_stormlands = {
            set_in_empire = yes
        }
        region_iron_islands = {
            set_in_empire = yes
        }
        region_reach = {
            set_in_empire = yes
        }
        region_westerlands = {
            set_in_empire = yes
        }
        region_dorne = {
            set_in_empire = yes
        }
        
        # Remove HRE for Shattered World
        if = {
            limit = { NOT = { is_year = 101 } }
            dismantle_hre = yes
        }
        
        # Bookmark flags
        if = {
			limit = {
				is_year = 106
				NOT = { is_year = 106 }
			}
			set_global_flag = war_for_the_stepstones
		}
		
		if = {
			limit = {
				is_year = 129
				NOT = { is_year = 129 }
			}
			set_global_flag = the_dance_of_dragons
		}
		
		if = {
			limit = {
				is_year = 282
				NOT = { is_year = 282 }
			}
			set_global_flag = roberts_rebellion
		}
		
		if = {
			limit = {
				is_year = 900
				NOT = { is_year = 900 }
			}
			set_global_flag = shattered_world
		}
	}
}
