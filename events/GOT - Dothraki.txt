# Dothraki Succession Crisis
country_event = {
	id = GOT_DOTHRAKI.1
	title = GOT_DOTHRAKI.1.title
	desc = GOT_DOTHRAKI.1.desc
	picture = KING_SICK_IN_BED_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		government = khas
        NOT = { has_country_flag = dothraki_daenerys }
		NOT = { has_country_flag = no_new_succession_crisis_if_pretender_rebels }
		num_of_cities = 4
		OR = {
			NOT = { legitimacy_or_horde_unity = 50 }
			has_regency = yes
		}
	}
	
	immediate = {
		hidden_effect = {
			if = {
				limit = {
					NOT = { legitimacy_or_horde_unity = 50 }
				}
				random_owned_province = {
					limit = {
						is_capital = no
					}
					spawn_rebels = {
						type = pretender_rebels
						size = 2
						win = yes
					}
				}
			}
		}
	}
	
	option = {
		name = GOT_DOTHRAKI.1.option.a
		trigger = {
			NOT = { legitimacy_or_horde_unity = 50 }
		}
		if = {
			limit = {
				any_owned_province = {
					NOT = { is_core = ROOT }
					can_spawn_rebels = nationalist_rebels
				}
			}
			hidden_effect = {
				every_owned_province = {
					limit = {
						NOT = { is_core = ROOT }
						can_spawn_rebels = nationalist_rebels
					}
					nationalist_rebels = 1
				}
			}
			custom_tooltip = tribal_succession_crisis_non_cores_will_revolt
		}
		add_ruler_modifier = {
			name = "dothraki_succession_crisis"
		}
		custom_tooltip = tribal_crisis_can_end_when
	}
	
	option = {
		name = GOT_DOTHRAKI.1.option.b
		trigger = {
			has_regency = yes
		}
		define_ruler = {
			dynasty = ROOT #Uncles, or other relatives were by far the most likely coup-makers
		}
		kill_heir = yes
		add_horde_unity = -25
	}
}

# A New Khal!
# The ascension of [Root.Monarch.GetName] as khal has upset several of the kos within the khalasar. They threaten to split the khalasar unless [Root.Monarch.GetTitle] [Root.Monarch.GetName] proves his strength.
country_event = {
	id = GOT_DOTHRAKI.5
	title = GOT_DOTHRAKI.5.title
	desc = GOT_DOTHRAKI.5.desc
	picture = HORDE_ON_HORSEBACK_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		government = khas
        NOT = { has_country_flag = dothraki_daenerys }
	}
	
	# To war!
	option = {
		name = GOT_DOTHRAKI.5.option.a
		
		trigger = {
			any_neighbor_country = {
				NOT = { war_with = ROOT }
			}
		}
		
		random_neighbor_country = {
			ROOT = {
				declare_war_with_cb = {
					who = PREV
					casus_belli = cb_horde_vs_civ
				}
			}
		}
		
        add_ruler_modifier = {
            name = strong_khal
        }
        
		ai_chance = {
			factor = 50
		}
	}
	
	# [Root.Monarch.GetName] fights [Root.Monarch.GetHerHis] kos!
	option = {
		name = GOT_DOTHRAKI.5.option.b
		
		add_prestige = 5
		
		random_list = {
            50 = {
                add_ruler_modifier = {
                    name = strong_khal
                }
            }
            50 = {
                add_ruler_modifier = {
                    name = weak_khal
                }
            }
        }
		
		ai_chance = {
			factor = 40
		}
	}
	
	# Ignore their threat
	option = {
		name = GOT_DOTHRAKI.5.option.c
		
		add_ruler_modifier = {
            name = weak_khal
        }
		
		ai_chance = {
			factor = 10
		}
	}
}

	id = GOT_DOTHRAKI.6
	title = GOT_DOTHRAKI.6.title
	desc = GOT_DOTHRAKI.6.desc

		NOT = { tag = REB }
		FROM = {
            is_capital = yes

				NOT = { has_province_flag = was_culled }
				had_province_flag = { flag = was_culled days = 3650 }
			}
		}
	}

		hidden_effect = {
			FROM = {
				set_province_flag = was_culled

			}
		}
	}
	
	is_triggered_only = yes

		days = 1
	}

			factor = 75

				factor = 2
				NOT = { years_of_income = 1 }
			}
		}

		if = {
            limit = { 
                FROM = {
                    development = 50
                }
            }
            add_treasury = 500
        }
        else_if = {
            limit = { 
                FROM = {
                    development = 40
                    NOT = { development = 50 }
                }
            }
            add_treasury = 400
        }
        else_if = {
            limit = { 
                FROM = {
                    development = 30
                    NOT = { development = 40 }
                }
            }

        }

            limit = { 
                FROM = {

                    NOT = { development = 30 }
                }
            }
            add_treasury = 200
        }

            limit = {

                }
            }

                }
            }
            add_horde_unity = -25
        }
        else_if = {
            limit = { 
                FROM = {
                    development = 30
                    NOT = { development = 40 }
                }
            }
            add_horde_unity = -20
        }
        else_if = {

# Delayed Payments
country_event = {
	id = GOT_DOTHRAKI.9
	title = GOT_DOTHRAKI.9.title
	desc = GOT_DOTHRAKI.9.desc
	picture = NOT_DISCIPLINED_SOLDIERS_eventPicture

	trigger = {
		is_at_war = yes
		horde_unity = 50
		any_neighbor_country = {
			war_with = ROOT
			any_owned_province = {
				controlled_by = ROOT
			}
		}
	}
	
	immediate = {
		hidden_effect = {
			random_neighbor_country = {
				limit = {
					war_with = ROOT
					any_owned_province = {
						controlled_by = ROOT
					}
				}
				save_event_target_as = enemy_country
				random_owned_province = {
					limit = {
						controlled_by = ROOT
					}
					save_event_target_as = delayed_payments_province
				}
			}
		}
	}
	
	mean_time_to_happen = {
		months = 500
	}
	

	option = {
		name = GOT_DOTHRAKI.9.option.a
        
		ai_chance = {
			factor = 75
		}
        
		add_yearly_manpower = -1.5
	}
	option = {
		name = GOT_DOTHRAKI.9.option.b
        
		ai_chance = {
			factor = 25
		}
        
		add_horde_unity = -10
        
		event_target:delayed_payments_province = {
			area = {
				limit = {
					OR = {
						controlled_by = ROOT
						AND = {
							controlled_by = event_target:enemy_country
							any_neighbor_province = {
								controlled_by = ROOT
							}
						}
					}
				}
				add_devastation = 45
			}
		}
	}
}
