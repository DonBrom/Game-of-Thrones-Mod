namespace = GOT_ROYAL_HOUSE

# The Iron Throne
country_event = {
	id = GOT_ROYAL_HOUSE.1
	title = GOT_ROYAL_HOUSE.1.title
	desc = GOT_ROYAL_HOUSE.1.desc
	picture = GOOD_WITH_MONARCH_eventPicture
	
	is_triggered_only = yes
	
	major = yes
	
	trigger = {
        OR = {
            has_reform = daenerys_host_reform
            has_reform = noble_house_reform
        }
        FROM = { has_reform = royal_house_reform }
	}
	
	option = {
		name = "GOT_ROYAL_HOUSE.1.option.a"
		
        add_prestige = 100
        
        if = {
            limit = {
                has_reform = noble_house_reform
            }
            remove_government_reform = noble_house_reform
        }
        if = {
            limit = {
                has_reform = daenerys_host_reform
            }
            remove_government_reform = daenerys_host_reform
        }
        
        add_government_reform = royal_house_reform
        
        hidden_effect = { set_global_flag = update_royal_subjects } # Updates royal subjects
        
        # All noble houses will be upset by the usurper
        every_country = {
            limit = {
                has_reform = noble_house_reform
            }
            add_opinion = {
                who = ROOT
                modifier = usurper
            }
        }
	}
}
