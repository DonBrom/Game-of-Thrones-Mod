namespace = slaves

# Setup
country_event = {
	id = slaves.100
	title = slaves.100.title
	desc = slaves.100.desc
	picture = FEAST_eventPicture
	
	is_triggered_only = yes
	hidden = yes
    
    trigger = {
        NOT = { has_country_flag = ran_slave_setup }
    }
    
    immediate = {   
        set_country_flag = ran_slave_setup 
        
        set_variable = {
            which = slave_count
            value = 0
        }
        
        if = {
            limit = {
				NOT = { tag = U25 }
                capital_scope = {
                    NOT = { continent = westeros }
                }
            }
            add_country_modifier = {
                name = slavery_permitted
                duration = -1
                hidden = yes
            }
        }
        if = {
            limit = {
				OR = {
					tag = U25
					capital_scope = {
						continent = westeros
					}
				
				}
				
            }
            add_country_modifier = {
                name = slavery_prohibited
                duration = -1
                hidden = yes
            }
        }
    }
    
	option = {
		name = slaves.1.option.a
		
	}
}
