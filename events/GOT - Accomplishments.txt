# On Actions
# History is written by the victors! - MIL
# War won
country_event = {
	id = GOT_ACCOMPLISHMENTS.1
	title = "GOT_ACCOMPLISHMENTS.1.title"
	desc = "GOT_ACCOMPLISHMENTS.1.desc"
	picture = BATTLE_eventPicture

	is_triggered_only = yes

	trigger = {
		NOT = { has_country_modifier = mil_tech_001 }
	}
	
	option = {
		name = "GOT_ACCOMPLISHMENTS.1.option.a"
		
		hidden_effect = {
			add_country_modifier = {
				name = mil_tech_001
				duration = 3650
				hidden = yes
			}
		}
		
		add_mil_power = 50
	}
}

# Colonial Vigor - DIP
# 2+ colonies, scaling
country_event = {
	id = GOT_ACCOMPLISHMENTS.7
	title = "GOT_ACCOMPLISHMENTS.7.title"
	desc = "GOT_ACCOMPLISHMENTS.7.desc"
	picture = COLONIZATION_eventPicture

	is_triggered_only = yes
	
	trigger = {
		NOT = { has_country_modifier = dip_tech_003 }
		
		num_of_colonies = 2
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = "GOT_ACCOMPLISHMENTS.7.option.a"
		
		hidden_effect = {
			add_country_modifier = {
				name = dip_tech_003
				duration = 3650
				hidden = yes
			}
		}
		
		if = {
			limit = { 
				num_of_colonies = 2
				NOT = { num_of_colonies = 3 } 
			}
			add_dip_power = 50
		}
		
		if = {
			limit = { 
				num_of_colonies = 3
				NOT = { num_of_colonies = 4 } 
			}
			add_dip_power = 75
		}
		
		if = {
			limit = { 
				num_of_colonies = 4
			}
			add_dip_power = 100
		}
	}
}
