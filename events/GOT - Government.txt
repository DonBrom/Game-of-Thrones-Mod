#title: [Root.Monarch.GetTitle] of [Root.GetName]
#desc: The previous [Root.Monarch.GetTitle] has passed away, so it is time for [Root.GetName] to pick a new [Root.Monarch.GetTitle]. However, if the people think ill of the chosen [Root.Monarch.GetTitle], there is likely to be pretenders to the position.
#1: A candidate from the Tiger Party
#2: A candidate from the Elephant Party
#3: Re-elect [Root.Monarch.GetName]
country_event = {
	id = GOT_GOVERNMENT.2
	title = GOT_GOVERNMENT.2.title
	desc = GOT_GOVERNMENT.2.desc
	picture = IMPORTANT_STATUE_eventPicture
	
	is_triggered_only = yes

	trigger = {
		is_free_city_government = yes
		primary_culture = volantene
	}
	
	# Tiger
	option = {
		name = GOT_GOVERNMENT.2.option.a

		clr_country_flag = has_tiger_ruler
		clr_country_flag = has_elephant_ruler
		
		define_ruler = {
			adm = 3
			dip = 1
			mil = 4
			fixed = yes
		}
		
		add_estate_loyalty = {
			estate = estate_elephants
			loyalty = -5
		}
		
		set_country_flag = has_tiger_ruler
		
		ai_chance = {
			factor = 100
		}
	}
	
	# Elephant
	option = {
		name = GOT_GOVERNMENT.2.option.b
		
		clr_country_flag = has_tiger_ruler
		clr_country_flag = has_elephant_ruler
		
		define_ruler = {
			adm = 3
			dip = 4
			mil = 1
			fixed = yes
		}
		
		add_estate_loyalty = {
			estate = estate_tigers
			loyalty = -5
		}
		
		set_country_flag = has_elephant_ruler
		
		ai_chance = {
			factor = 100
		}
	}
	
	# Re-elect
	option = {
		name = GOT_GOVERNMENT.2.option.c
		
		trigger = {
			OR = {
				has_country_flag = has_tiger_ruler
				has_country_flag = has_elephant_ruler
			}
		}
		
		random_list = {
			40 = {
			
			}
			20 = {
				change_adm = 1
			}
			20 = {
				change_dip = 1
			}
			20 = {
				change_mil = 1
			}
		}
		
		add_republican_tradition = -2
		
		custom_tooltip = RE_ELECTION_TOOLTIP
		
		ai_chance = {
			factor = 50
		}
	}
}
