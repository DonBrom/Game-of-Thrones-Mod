namespace = GOT_ESTATES

# A Noble Predicament
country_event = {
	id = GOT_ESTATES.1
	title = GOT_ESTATES.1.title
	desc = GOT_ESTATES.1.desc
	picture = DIPLOMACY_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		has_dlc = "The Cossacks"
		has_estate = estate_nobles
        NOT = {
            has_estate_influence_modifier = {
                estate = estate_nobles
                modifier = EST_VAL_RESOLVED_PREDICAMENT
            }
        }
        NOT = {
            has_estate_influence_modifier = {
                estate = estate_nobles
                modifier = EST_VAL_PREDICAMENT_INFLUENCE
            }
        }
	}
	
	option = {
		name = GOT_ESTATES.1.option.a
		
		add_estate_loyalty = {
			estate = estate_nobles
			loyalty = 15
		}
		
		add_estate_influence_modifier = {
			estate = estate_nobles
			desc = EST_VAL_RESOLVED_PREDICAMENT
			influence = 15
			duration = 7200
		}
		
		ai_chance = {
			factor = 50
			modifier = {
				factor = 0
				
				estate_influence = {
					estate = estate_nobles
					influence = 60
				}
			}
		}
	}
	
	option = {
		name = GOT_ESTATES.1.option.b
		
		add_estate_loyalty = {
			estate = estate_nobles
			loyalty = -15
		}
		
		add_estate_influence_modifier = {
			estate = estate_nobles
			desc = EST_VAL_PREDICAMENT_INFLUENCE
			influence = -15
			duration = 7200
		}
		
		ai_chance = {
			factor = 50
		}
	}
}
