# Merchants Resented
country_event = {
	id = got_overextension.6
	title = got_overextension.6.title
	desc = got_overextension.6.desc
	picture = MERCHANTS_TALKING_eventPicture

	is_triggered_only = yes
	
	trigger = {
		num_of_merchants = 1
		any_active_trade_node = {
			NOT = { 
				has_trade_modifier = {
					who = ROOT
					name = merchant_recalled
				}
			}	
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}

	option = {
		name = got_overextension.6.option.a
		random_active_trade_node = {
			limit = {
				NOT = { 
					has_trade_modifier = {
						who = ROOT
						name = merchant_recalled
					}
				}				
			}
			if = {
				limit = {
					trade_share = {
						country = ROOT
						share = 5
					}
				}
				add_trade_node_income = -10
			}
			add_trade_modifier = {
				who = root
				duration = 3650
				power = -10
				key = merchant_recalled
			}
		}
		
		if = {
			limit = { has_estate = estate_merchants }
						
			add_estate_loyalty = {
				estate = estate_merchants
				loyalty = -5
			}
		}
	}
}

# Legitimacy Questioned
country_event = {
	id = got_overextension.8
	title = got_overextension.8.title
	desc = got_overextension.8.desc
	picture = COURT_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		legitimacy_or_horde_unity = 15
		is_monarchy_government = yes
		is_lesser_in_union = no
	}

	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = got_overextension.8.option.a

		if = {
			limit = { 
                OR = {
                    has_reform = khalasar_reform
                    has_reform = jhattar_reform
                }
            }
			add_horde_unity = -15
		}
		else = {
            add_legitimacy = -15
        }
        
		if = {
			limit = { has_estate = estate_nobles }
						
			add_estate_loyalty = {
				estate = estate_nobles
				loyalty = -5
			}
		}
	}
}
