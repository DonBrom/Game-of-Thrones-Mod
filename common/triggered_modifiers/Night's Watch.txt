# Night's Watch
# Applied if the Night's Watch isn't massive (1000 dev+)
the_nights_watch_supported = {
	potential = {
		tag = A02
		government = nights_watch
        NOT = { total_development = 1000 }
	}
	
	trigger = {
        any_country = {
            culture_group = culture_westeros
            government = westeros_monarchy
            has_opinion = {
                who = ROOT
                value = 175
            }
        }
	}
	
	global_regiment_recruit_speed = -0.05
    land_forcelimit_modifier = 0.05
    land_maintenance_modifier = -0.05
    manpower_recovery_speed = 0.05
}
