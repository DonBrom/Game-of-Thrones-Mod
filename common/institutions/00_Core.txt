# Westerosi
westerosi_practices = {
    penalty = 0.05
    
    bonus = {
        global_manpower_modifier = 0.1
	}

	history = {
        owner = {
            OR = {
                technology_group = western
                technology_group = eastern
            }
        }
	}
	
	can_embrace = {
		always = yes
	}
    
    embracement_speed = {
        # Regular friendly border embracement
        modifier = {
			factor = 0.5
			scale = yes
            
            custom_trigger_tooltip = {
				tooltip = tt_friendly_border_westerosi
                
                any_friendly_coast_border_province = {
                    westerosi_practices = 100
                }
            }
        }
        
        # High relations will slowly increase embracement
        modifier = {
			factor = 0.5
			scale = yes
            
            custom_trigger_tooltip = {
				tooltip = tt_high_relations_westerosi
                
                owner = {
                    any_known_country = {
                        capital_scope = { westerosi_practices = 100 }
                        has_opinion = {
                            who = PREV
                            value = 150
                        }
                    }
                }
            }
        }
        
        # Has Embraced
        modifier = {
			factor = 2
			scale = yes
            
            potential = {
				owner = {
					has_institution = westerosi_practices
				}
			}
			custom_trigger_tooltip = {
				tooltip = tt_embraced_westerosi
                
				owner = {
					has_institution = westerosi_practices
				}
			}
        }
    }
    
    ai_will_do = {
		factor = 24
		
		modifier = {
			factor = 0.25
			is_at_war = yes
		}
		modifier = {
			factor = 2
			neighbors_tech_discount = 0.1
		}
		modifier = {
			factor = 2
			neighbors_tech_discount = 0.2
		}
		modifier = {
			factor = 2
			neighbors_tech_discount = 0.3
		}
		modifier = {
			factor = 2
			neighbors_tech_discount = 0.4
		}
	}
}
