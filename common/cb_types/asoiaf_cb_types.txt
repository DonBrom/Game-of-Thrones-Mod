    valid_for_subject = no

		is_part_of_hre = yes
		government = kingdom
        government_rank = 2
        FROM = { 
            is_emperor = yes 
            is_part_of_hre = yes
            government = kingdom
        }
        OR = { 
            ai = no

        }

}
