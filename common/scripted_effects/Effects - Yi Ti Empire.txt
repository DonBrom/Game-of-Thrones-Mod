		limit = {
			has_dlc = "Mandate of Heaven"
			is_emperor_of_china = yes
		}

dismantle_yi_ti_empire_effect = {
    FROM = {
		if = {
			limit = { is_emperor_of_china = yes }
			dismantle_empire_of_china = yes
			set_global_flag = empire_of_yiti_dismantled
		}
	}
}

grant_mandate_of_heaven_effect = {
    set_government_rank = 10
	set_mandate = 60
	region_yi_ti = {
		limit = {
			NOT = { is_core = ROOT }
		}
		add_permanent_claim = ROOT
	}
    if = {
		limit = {
			has_country_modifier = lost_mandate_of_heaven
		}
		remove_country_modifier = lost_mandate_of_heaven
	}
    country_event = {
		id = GOT_YITI_EMPIRE.2
		days = 1825
	}
}

remove_mandate_of_heaven_effect = {
	set_government_rank = 2
	add_country_modifier = {
		name = lost_mandate_of_heaven
		duration = 7300
	}
	add_stability = -2
	region_yi_ti = {
		limit = {
			is_permanent_claim = ROOT
		}
		remove_claim = ROOT
	}
}
