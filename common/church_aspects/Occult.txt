# Occult Inspiration
occult_inspiration_action = {
	sprite = "GFX_icon_enforce_rule_action"
	cost = 200
    
	potential = {
		has_dlc = "Rights of Man"
	}
	trigger = {
		always = yes
	}
	effect = {
        custom_tooltip = OCCULT_INSPIRATION_TT
        hidden_effect = {
            random_list = {
                33 = { add_adm_power = 500 }
                33 = { add_dip_power = 500 }
                33 = { add_mil_power = 500 }
            }
        }
	}
	ai_will_do = {
		factor = 1
	}
}
