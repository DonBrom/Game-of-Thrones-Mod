on_war_won = {
    if = { limit = { has_global_flag = enable_onaction_logging } log = "ON_ACTIONS - winner : [Root.GetName] - loser : [From.GetName] - on_war_won" }
	
    events = {
        GOT_ROYAL_EDICTS.2
    }
    
	random_events = {
		100 = GOT_ACCOMPLISHMENTS.1 # History is written by the victors!
		5000 = 0
	}
    
	# Mission Flag for Northmen Mission
	if = {
		limit = {
			has_country_flag = northmen_can_take_establish_dominance_mission
			OR = {
				primary_culture = men_of_the_north
				primary_culture = skagosi
				primary_culture = crannogmen
			}
			ROOT = {
				is_rival = FROM
			}
		}
		clr_country_flag = northmen_can_take_establish_dominance_mission
		set_country_flag = northmen_won_war_against_rival
	}
}

	if = { limit = { has_global_flag = enable_onaction_logging } log = "ON_ACTIONS - [Root.GetName] - on_raid_coast" }
    
    # Give Pirates CB to targets of Raid Coast
    FROM = {
        every_neighbor_province = {
            limit = { is_sea = no }
            owner = {
                if = {
                    limit = {
                        NOT = {
                            has_casus_belli = {
                                type = cb_pirates
                                target = ROOT
                            }
                        }
                    }
                    
                    add_casus_belli = {
                        target = ROOT
                        type = cb_pirates
                        months = 120
                    }
                }
            }
        }
    }
}
